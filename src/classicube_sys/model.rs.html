6#[macro_export]
7macro_rules! BoxDesc_Dim {
8    ($p1:expr, $p2:expr) => {
9        {
10            if $p1 < $p2 {
11                $p2 - $p1
12            } else {
13                $p1 - $p2
14            }
15        } as u8
16    };
17}

82        (
83            $crate::BoxDesc_Dims!($x1, $y1, $z1, $x2, $y2, $z2),
